# Aim: Producer-Consumer using Semaphores

import threading
import time
import random

buffer = []
buffer_size = 5

empty = threading.Semaphore(buffer_size)
full = threading.Semaphore(0)
mutex = threading.Semaphore(1)

def producer():
    while True:
        item = random.randint(1, 100)
        empty.acquire()
        mutex.acquire()

        buffer.append(item)
        print(f"Produced: {item} | Buffer: {buffer}")

        mutex.release()
        full.release()
        time.sleep(1)

def consumer():
    while True:
        full.acquire()
        mutex.acquire()

        item = buffer.pop(0)
        print(f"Consumed: {item} | Buffer: {buffer}")

        mutex.release()
        empty.release()
        time.sleep(1)

threading.Thread(target=producer).start()
threading.Thread(target=consumer).start()







# Aim: Reader-Writer using Semaphores

import threading
import time

read_count = 0
mutex = threading.Semaphore(1)
write_lock = threading.Semaphore(1)

def reader(id):
    global read_count
    while True:
        mutex.acquire()
        read_count += 1
        if read_count == 1:
            write_lock.acquire()
        mutex.release()

        print(f"Reader {id} is reading")
        time.sleep(1)

        mutex.acquire()
        read_count -= 1
        if read_count == 0:
            write_lock.release()
        mutex.release()

        time.sleep(1)

def writer(id):
    while True:
        write_lock.acquire()
        print(f"Writer {id} is writing")
        time.sleep(2)
        write_lock.release()
        time.sleep(1)

# Threads
for i in range(2):
    threading.Thread(target=reader, args=(i,)).start()

for i in range(2):
    threading.Thread(target=writer, args=(i,)).start()









# Aim: Dining Philosophers using Semaphores (Phil 1 to Phil 5)

import threading
import time

N = 5
forks = [threading.Semaphore(1) for _ in range(N)]

def philosopher(i):
    while True:
        print(f"Phil {i+1} is thinking")
        time.sleep(1)

        forks[i].acquire()
        forks[(i+1) % N].acquire()

        print(f"Phil {i+1} is eating")
        time.sleep(1)

        forks[i].release()
        forks[(i+1) % N].release()

# Create threads
for i in range(N):
    threading.Thread(target=philosopher, args=(i,)).start()






    # Aim: Implement Sleeping Barber Problem using Semaphores

import threading
import time
import random

# Number of waiting chairs
chairs = 3

# Semaphores
customers = threading.Semaphore(0)   # Customers waiting
barber = threading.Semaphore(0)      # Barber ready
mutex = threading.Semaphore(1)       # Mutual exclusion

waiting = 0  # Number of customers waiting

# Barber function
def barber_func():
    global waiting
    while True:
        customers.acquire()   # Wait for customer
        mutex.acquire()

        waiting -= 1
        print("Barber is cutting hair... Waiting:", waiting)

        barber.release()      # Barber is ready
        mutex.release()

        time.sleep(random.uniform(1, 2))  # Haircut time


# Customer function
def customer_func(id):
    global waiting

    mutex.acquire()
    if waiting < chairs:
        waiting += 1
        print(f"Customer {id} is waiting. Waiting:", waiting)

        customers.release()   # Notify barber
        mutex.release()

        barber.acquire()      # Wait for barber
        print(f"Customer {id} is getting haircut")

    else:
        print(f"Customer {id} left (No chairs available)")
        mutex.release()


# Create barber thread
threading.Thread(target=barber_func, daemon=True).start()

# Create customer threads
for i in range(10):
    time.sleep(random.uniform(0.5, 1.5))
    threading.Thread(target=customer_func, args=(i,)).start()
